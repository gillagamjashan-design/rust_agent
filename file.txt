import socket
import threading
import os
import mimetypes
from datetime import datetime
from pathlib import Path

class MultiThreadedWebServer:
    def __init__(self, host='127.0.0.1', port=8080, document_root='.'):
        self.host = host
        self.port = port
        self.document_root = Path(document_root).resolve()
        self.server_socket = None
        self.running = False
        
    def start(self):
        """Start the web server"""
        self.server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.server_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        self.server_socket.bind((self.host, self.port))
        self.server_socket.listen(5)
        self.running = True
        
        print(f"Server started at http://{self.host}:{self.port}")
        print(f"Document root: {self.document_root}")
        print("Press Ctrl+C to stop the server")
        
        try:
            while self.running:
                client_socket, client_address = self.server_socket.accept()
                print(f"Connection from {client_address}")
                
                # Create a new thread to handle the client
                client_thread = threading.Thread(
                    target=self.handle_client,
                    args=(client_socket, client_address)
                )
                client_thread.daemon = True
                client_thread.start()
        except KeyboardInterrupt:
            print("\nShutting down server...")
        finally:
            self.stop()
    
    def handle_client(self, client_socket, client_address):
        """Handle individual client connection"""
        try:
            # Receive the request
            request = client_socket.recv(4096).decode('utf-8')
            
            if not request:
                return
            
            # Parse the request
            lines = request.split('\n')
            if len(lines) == 0:
                return
                
            request_line = lines[0].strip()
            parts = request_line.split()
            
            if len(parts) < 2:
                self.send_error(client_socket, 400, "Bad Request")
                return
            
            method = parts[0]
            path = parts[1]
            
            print(f"[{threading.current_thread().name}] {method} {path}")
            
            if method == 'GET':
                self.handle_get(client_socket, path)
            else:
                self.send_error(client_socket, 501, "Method Not Implemented")
                
        except Exception as e:
            print(f"Error handling client {client_address}: {e}")
            try:
                self.send_error(client_socket, 500, "Internal Server Error")
            except:
                pass
        finally:
            client_socket.close()
    
    def handle_get(self, client_socket, path):
        """Handle GET requests"""
        # Remove query string if present
        path = path.split('?')[0]
        
        # Normalize path
        if path == '/':
            path = '/index.html'
        
        # Construct file path
        file_path = self.document_root / path.lstrip('/')
        
        try:
            # Security check: ensure the path is within document root
            file_path = file_path.resolve()
            if not str(file_path).startswith(str(self.document_root)):
                self.send_error(client_socket, 403, "Forbidden")
                return
            
            # Check if file exists and is a file
            if not file_path.exists():
                self.send_error(client_socket, 404, "Not Found")
                return
            
            if file_path.is_dir():
                # Try to serve index.html from directory
                index_file = file_path / 'index.html'
                if index_file.exists():
                    file_path = index_file
                else:
                    self.send_directory_listing(client_socket, file_path)
                    return
            
            # Read and send the file
            self.send_file(client_socket, file_path)
            
        except Exception as e:
            print(f"Error serving file {path}: {e}")
            self.send_error(client_socket, 500, "Internal Server Error")
    
    def send_file(self, client_socket, file_path):
        """Send a file to the client"""
        # Determine content type
        content_type, _ = mimetypes.guess_type(str(file_path))
        if content_type is None:
            content_type = 'application/octet-stream'
        
        # Read file
        with open(file_path, 'rb') as f:
            content = f.read()
        
        # Build response
        response = f"HTTP/1.1 200 OK\r\n"
        response += f"Content-Type: {content_type}\r\n"
        response += f"Content-Length: {len(content)}\r\n"
        response += f"Server: MultiThreadedWebServer/1.0\r\n"
        response += f"Date: {datetime.utcnow().strftime('%a, %d %b %Y %H:%M:%S GMT')}\r\n"
        response += "\r\n"
        
        client_socket.sendall(response.encode('utf-8'))
        client_socket.sendall(content)
    
    def send_directory_listing(self, client_socket, dir_path):
        """Send a directory listing"""
        relative_path = dir_path.relative_to(self.document_root)
        
        html = f"""<!DOCTYPE html>
<html>
<head>
    <title>Directory Listing: /{relative_path}</title>
    <style>
        body {{ font-family: Arial, sans-serif; margin: 20px; }}
        h1 {{ color: #333; }}
        ul {{ list-style: none; padding: 0; }}
        li {{ padding: 5px; }}
        a {{ text-decoration: none; color: #0066cc; }}
        a:hover {{ text-decoration: underline; }}
    </style>
</head>
<body>
    <h1>Directory Listing: /{relative_path}</h1>
    <ul>
"""
        
        # Add parent directory link if not at root
        if dir_path != self.document_root:
            html += f'        <li><a href="../">📁 ..</a></li>\n'
        
        # List directories first, then files
        items = sorted(dir_path.iterdir(), key=lambda x: (not x.is_dir(), x.name))
        
        for item in items:
            icon = "📁" if item.is_dir() else "📄"
            name = item.name + ("/" if item.is_dir() else "")
            html += f'        <li><a href="{name}">{icon} {name}</a></li>\n'
        
        html += """    </ul>
</body>
</html>"""
        
        content = html.encode('utf-8')
        
        response = f"HTTP/1.1 200 OK\r\n"
        response += f"Content-Type: text/html; charset=utf-8\r\n"
        response += f"Content-Length: {len(content)}\r\n"
        response += f"Server: MultiThreadedWebServer/1.0\r\n"
        response += "\r\n"
        
        client_socket.sendall(response.encode('utf-8'))
        client_socket.sendall(content)
    
    def send_error(self, client_socket, status_code, message):
        """Send an error response"""
        html = f"""<!DOCTYPE html>
<html>
<head>
    <title>{status_code} {message}</title>
</head>
<body>
    <h1>{status_code} {message}</h1>
    <hr>
    <p>MultiThreadedWebServer/1.0</p>
</body>
</html>"""
        
        content = html.encode('utf-8')
        
        response = f"HTTP/1.1 {status_code} {message}\r\n"
        response += f"Content-Type: text/html; charset=utf-8\r\n"
        response += f"Content-Length: {len(content)}\r\n"
        response += f"Server: MultiThreadedWebServer/1.0\r\n"
        response += "\r\n"
        
        try:
            client_socket.sendall(response.encode('utf-8'))
            client_socket.sendall(content)
        except:
            pass
    
    def stop(self):
        """Stop the server"""
        self.running = False
        if self.server_socket:
            self.server_socket.close()
        print("Server stopped")
